To process all the files and produce a suitable output for the next stage, I'll write a Python script that reads each file, extracts the relevant information, and stores it in a list of dictionaries.

```python
import csv
from datetime import datetime

# Define a function to parse a CSV file
def parse_csv(file_path):
    with open(file_path, 'r') as file:
        reader = csv.DictReader(file)
        data = []
        for row in reader:
            data.append({
                'Supplier': row['Supplier'],
                'InvoiceNumber': row['InvoiceNumber'],
                'InvoiceDate': datetime.strptime(row['InvoiceDate'], '%Y-%m-%d').strftime('%Y-%m-%d'),
                'DueDate': datetime.strptime(row['DueDate'], '%Y-%m-%d').strftime('%Y-%m-%d'),
                'Subtotal': float(row['Subtotal']),
                'Tax': float(row['Tax']),
                'Total': float(row['Total']),
                'Currency': row['Currency'],
                'PO': row['PO'],
                'PaymentTerms': row['PaymentTerms']
            })
    return data

# Define a function to parse a TXT file
def parse_txt(file_path):
    with open(file_path, 'r') as file:
        lines = file.readlines()
        data = []
        for i in range(0, len(lines), 7):
            data.append({
                'Supplier': lines[i].strip().split(':')[1],
                'InvoiceNumber': lines[i+1].strip().split(':')[1],
                'InvoiceDate': datetime.strptime(lines[i+2].strip(), '%Y-%m-%d').strftime('%Y-%m-%d'),
                'DueDate': datetime.strptime(lines[i+3].strip(), '%Y-%m-%d').strftime('%Y-%m-%d'),
                'Subtotal': float(lines[i+4].strip().split(':')[1]),
                'Tax': float(lines[i+5].strip().split(':')[1]),
                'Total': float(lines[i+6].strip()),
                'Currency': lines[0].strip().split(':')[2],
                'PO': lines[7].strip().split(':')[1],
                'PaymentTerms': lines[8].strip()
            })
    return data

# Define a function to parse all files
def parse_files():
    files = [
        # CSV files
        ('invoice_sample_01.csv', parse_csv),
        ('invoice_sample_02.csv', parse_csv),
        ('invoice_sample_03.csv', parse_csv),
        ('invoice_sample_04.csv', parse_csv),
        ('invoice_sample_05.csv', parse_csv),
        ('invoice_sample_06.csv', parse_csv),
        ('invoice_sample_07.csv', parse_csv),
        ('invoice_sample_08.csv', parse_csv),
        ('invoice_sample_09.csv', parse_csv),
        ('invoice_sample_10.csv', parse_csv),

        # TXT files
        ('invoice_sample_01.txt', parse_txt),
        ('invoice_sample_02.txt', parse_txt),
        ('invoice_sample_03.txt', parse_txt),
        ('invoice_sample_04.txt', parse_txt),
        ('invoice_sample_05.txt', parse_txt),
        ('invoice_sample_06.txt', parse_txt),
        ('invoice_sample_07.txt', parse_txt),
        ('invoice_sample_08.txt', parse_txt),
        ('invoice_sample_09.txt', parse_txt),
        ('invoice_sample_10.txt', parse_txt)
    ]
    
    data = []
    for file_path, parser in files:
        data.extend(parser(file_path))
        
    return data

# Execute the function to parse all files
data = parse_files()
```

This script will output a list of dictionaries, where each dictionary represents an invoice with its corresponding fields. The `parse_csv` and `parse_txt` functions handle the parsing of CSV and TXT files, respectively.

Please note that this script assumes that the input files are in the correct format and that all required fields are present in each file. If there are any errors or inconsistencies in the input data, the script may not work as expected.
